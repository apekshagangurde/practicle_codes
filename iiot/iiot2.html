Python Code for Sending Data to ThingSpeak

import http.client
import urllib.parse

key = 'G1UN993WDMN29NFU'  # Replace with the actual ThingSpeak API key

t = 22  # Start temperature

while True:
    t += 1
    print("temperature", t)
    params = urllib.parse.urlencode({'field1': t, 'key': key})
    headers = {"Content-type": "application/x-www-form-urlencoded", "Accept": "text/plain"}
    con = http.client.HTTPConnection("api.thingspeak.com")
    con.request("POST", "/update", params, headers)
    response = con.getresponse()
    print(response.status, response.reason)
    data = response.read()
    con.close()
