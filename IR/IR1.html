import nltk
from nltk.corpus import stopwords
from nltk.stem import PorterStemmer
from nltk.tokenize import word_tokenize, sent_tokenize

# Sample random text (100 words)
random_text = """
Data processing encompasses a series of operations that convert raw data into structured
and organized information. This process begins with data collection, where data is gathered
from various sources such as sensors, databases, forms, or external systems. Once collected,
the data can be in various formats, including text, numbers, images, or multimedia.
The next step in data processing is data cleaning and validation. This involves identifying
and correcting errors, inconsistencies, and missing values in the data. Clean and accurate data
is essential for reliable analysis and decision-making. Data cleaning often involves techniques
like outlier detection and data imputation.
After data cleaning, data transformation is performed. This includes tasks like data normalization,
aggregation, and summarization. Normalization ensures that data is on a consistent scale, while
aggregation and summarization reduce data complexity by generating statistics or aggregating data into meaningful groups.
Data processing also includes data integration, where data from multiple sources is combined
into a unified dataset. Integration can be challenging due to differences in data structures and
formats. Techniques like data mapping and data warehousing are used to facilitate integration.
"""

# Tokenize the text into words
words = word_tokenize(random_text)

# Initialize the NLTK Porter Stemmer
stemmer = PorterStemmer()

# Get the English stop words
nltk.download('stopwords')
stop_words = set(stopwords.words("english"))

# Initialize a list to store the preprocessed words
preprocessed_words = []

# Perform text preprocessing
for word in words:
    # Remove punctuation and convert to lowercase
    word = word.lower()
    word = word.strip('.,?!-()[]{}"')

    # Check if the word is not a stop word
    if word not in stop_words:
        # Stem the word
        word = stemmer.stem(word)

        # Add the preprocessed word to the list
        preprocessed_words.append(word)

# Join the preprocessed words back into a text
preprocessed_text = " ".join(preprocessed_words)

# Print the original text and preprocessed text
print("Original Text:")
print(random_text)
print("\nPreprocessed Text:")
print(preprocessed_text)

Explanation of the Code:

import nltk: Imports the NLTK library, which is essential for natural language processing tasks.

from nltk.corpus import stopwords: Imports the list of stopwords, which are common words that are typically removed in text processing.

from nltk.stem import PorterStemmer: Imports the Porter Stemmer algorithm for stemming words, which reduces words to their root form.

from nltk.tokenize import word_tokenize, sent_tokenize: Imports functions for tokenizing text into words and sentences.

random_text = """...""": Defines a multi-line string containing sample text for preprocessing.

words = word_tokenize(random_text): Tokenizes the sample text into individual words.

stemmer = PorterStemmer(): Initializes the Porter Stemmer for stemming operations.

nltk.download('stopwords'): Downloads the stopwords dataset if it's not already downloaded.

stop_words = set(stopwords.words("english")): Retrieves the English stopwords and stores them in a set for fast lookup.

preprocessed_words = []: Initializes an empty list to store preprocessed words after cleaning and stemming.

for word in words:: Starts a loop to iterate through each word in the tokenized list.

word = word.lower(): Converts the word to lowercase to ensure uniformity.

word = word.strip('.,?!-()[]{}"'): Removes punctuation from the word.

if word not in stop_words:: Checks if the word is not a stop word before processing it further.

word = stemmer.stem(word): Applies stemming to the word to reduce it to its root form.

preprocessed_words.append(word): Adds the preprocessed word to the list of cleaned words.

preprocessed_text = " ".join(preprocessed_words): Joins the list of cleaned words back into a single string.

print("Original Text:"): Prints a label for the original text output.

print(random_text): Outputs the original sample text.

print("\nPreprocessed Text:"): Prints a label for the preprocessed text output.

print(preprocessed_text): Outputs the preprocessed text after cleaning and stemming.
